<script>
    var no = prompt("Enter the number", "");
    no = parseInt(no);

    if(no < 0)
        document.write("You entered Negative Number.");
    else if(no > 0)
        document.write("You entered Positive Number.");
    else
        document.write("You entered Zero.");
</script>
